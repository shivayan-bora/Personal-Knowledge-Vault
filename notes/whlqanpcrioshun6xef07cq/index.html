Mutations

Unlike queries, mutations are used to perform CRUD operations on data or perform server side-effects. For this, we can use a useMutation hook.

function App() {
   const mutation = useMutation(newTodo => {
     return axios.post('/todos', newTodo)
   })

   return (
     <div>
       {mutation.isLoading ? (
         'Adding todo...'
       ) : (
         <>
           {mutation.isError ? (
             <div>An error occurred: {mutation.error.message}</div>
           ) : null}

           {mutation.isSuccess ? <div>Todo added!</div> : null}
 
           <button
             onClick={() => {
               mutation.mutate({ id: new Date(), title: 'Do Laundry' })
             }}
           >
             Create Todo
           </button>
         </>
       )}
     </div>
   )
 }
